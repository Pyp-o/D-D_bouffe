@startuml
start
    repeat
        repeat
        :que voulez-vous faire ?;
        repeat while (choix impossible) i (yes)

        split
            :gestion equipement;
            :afficher equipement;
                split
                    :equiper;
                split again
                    :desequiper;
                 endsplit
        split again
            :utilisation consommables;
            :afficher consommables;
            :manger;

        split again
            if (dans cuisine) then (yes)
                :creer potions;
                :afficher ressources
                :combiner;
            endif
        endsplit
        repeat while(terminer) is (no)
    stop
@enduml
